def timeConversion(s):
    # Write your code here
    AM_PM = s[-2:]
    hours = int(s[:2])
    minutes_Seconds = s[2:-2]
    if AM_PM == 'AM':
        if hours == 12:
            hours = 0
    else :
        if hours!= 12:
            hours += 12
    return f"{hours:02d}{minutes_Seconds}" 

if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    s = input()

    result = timeConversion(s)

    fptr.write(result + '\n')

    fptr.close()
