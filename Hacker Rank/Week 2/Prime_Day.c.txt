#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int findPrimeDates(int d1, int m1, int y1, int d2, int m2, int y2) {
    int result = 0;
    int months[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

    while (d1 != d2 || m1 != m2 || y1 != y2) {
        int x = d1 * 1000000 + m1 * 10000 + y1; 
        if (x % 4 == 0 || x % 7 == 0) { 
            result++;
        }

        d1++;
        if (d1 > months[m1]) {
            if (m1 == 2 && ((y1 % 400 == 0) || (y1 % 4 == 0 && y1 % 100 != 0))) {
                if (d1 > 29) {
                    d1 = 1;
                    m1++;
                }
            } else {
                d1 = 1;
                m1++;
            }
        }
        if (m1 > 12) {
            m1 = 1;
            y1++;
        }
    }

    int x = d1 * 1000000 + m1 * 10000 + y1;
    if (x % 4 == 0 || x % 7 == 0) {
        result++;
    }

    return result;
}

int main() {
    int d1, m1, y1, d2, m2, y2;
    scanf("%d-%d-%d", &d1, &m1, &y1);
    scanf("%d-%d-%d", &d2, &m2, &y2);
    printf("%d\n", findPrimeDates(d1, m1, y1, d2, m2, y2));
    return 0;
}
